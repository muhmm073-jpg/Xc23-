    <script>
        lucide.createIcons();
        const apiKey = ""; // API Key

        // Mobil PWA Banner Simülasyonu
        if (window.innerWidth < 768) {
            setTimeout(() => {
                document.getElementById('installBanner').style.display = 'flex';
            }, 2000);

        function hideBanner() {
            document.getElementById('installBanner').style.display = 'none';
        }

        function startApp() {
            document.getElementById('landingView').style.display = 'none';
            document.getElementById('appView').style.display = 'flex';
        }

        function login() {
            document.getElementById('authOverlay').style.opacity = '0';
            setTimeout(() => {
                document.getElementById('authOverlay').style.display = 'none';
                document.getElementById('userName').innerText = "XC KULLANICISI #01";
            }, 500);
        }

        async function sendMessage() {
            const input = document.getElementById('userInput');
            const chatBox = document.getElementById('chatBox');
            const val = input.value.trim();
            if(!val) return;

            appendMsg(val, true);
            input.value = '';

            try {
                const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({
                        contents: [{ parts: [{ text: val }] }],
                        systemInstruction: { parts: [{ text: "Sen XC'sin. Yanıtların çok profesyonel, analitik ve verimli olsun. Mobil ve web uyumlu bir asistan olduğunu unutma." }] }
                    })
                });
                const data = await response.json();
                const aiText = data.candidates?.[0]?.content?.parts?.[0]?.text || "Veri hattında bir hata oluştu.";
                appendMsg(aiText, false);
            } catch (e) {
                appendMsg("Sistem çevrimdışı. Lütfen bağlantınızı kontrol edin.", false);
            }
        }

        function appendMsg(text, isUser) {
            const div = document.createElement('div');
            div.className = `flex gap-4 ${isUser ? 'flex-row-reverse' : ''}`;
            div.innerHTML = `
                <div class="w-8 h-8 rounded-lg ${isUser ? 'bg-gray-800' : 'bg-blue-600'} flex items-center justify-center shrink-0">
                    <i data-lucide="${isUser ? 'user' : 'zap'}" class="w-4 h-4 text-white"></i>
                </div>
                <div class="${isUser ? 'bg-blue-600' : 'app-card'} p-4 rounded-3xl ${isUser ? 'rounded-tr-none' : 'rounded-tl-none'} max-w-[85%] text-sm">
                    ${text}
                </div>
            `;
            chatBox.appendChild(div);
            chatBox.scrollTop = chatBox.scrollHeight;
            lucide.createIcons();
        }

        document.getElementById('userInput').addEventListener('keypress', (e) => {
            if(e.key === 'Enter') sendMessage();
        });
    </script>
